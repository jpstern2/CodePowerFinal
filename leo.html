def calculate_sentiment(text):
    # Run VADER on the text
    scores = sentimentAnalyser.polarity_scores(text)
    # Extract the compound score
    compound_score = scores['compound']
    # Return compound score

import requests
from bs4 import BeautifulSoup
# URL of the page to scrape
url = "https://en.wikipedia.org/wiki/Maritime_history"
response = requests.get(url)
# Parse the HTML content
document = BeautifulSoup(response.content, 'html.parser')
# Find the table with historical events
# Assume the events are listed in <h3> tags
events = document.find_all('h3')
# Extract the data
event_data = []
for event in events:
    year = event.find('a').text if event.find('a') else 'N/A'  # Extract the year if available
    description = event.text
    event_data.append({'Year': year, 'Event': description})
# Convert the data to a DataFrame
events_df = pd.DataFrame(event_data)
# Display the first few rows
print(events_df.head())
